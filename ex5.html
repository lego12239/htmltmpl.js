Example of using IFEQ, IFNEQ, IFGT, IFGE, IFLT, IFLE.
{%TMPL_IFEQ NAME=val1 VALUE="a"%}
val1 == "a"
{%/TMPL_IFEQ%} {%TMPL_IFNEQ NAME=val2 WITH=val1%}
val2 != val1
{%/TMPL_IFNEQ%} {%TMPL_IFEQ NAME=val1 WITH=val2%}
val1 == val2
{%TMPL_ELSE%}
val1 != val2
{%/TMPL_IFEQ%} {%TMPL_IFGT NAME=val3 VALUE=4%}
val3 > 4
{%/TMPL_IFGT%} {%TMPL_IFGE NAME=val3 VALUE=5%}
val3 >= 5
{%/TMPL_IFGE%} {%TMPL_IFLT NAME=val3 VALUE=6%}
val3 < 6
{%/TMPL_IFLT%} {%TMPL_IFLE NAME=val3 VALUE=5%}
val3 <= 5
{%/TMPL_IFLE%}
